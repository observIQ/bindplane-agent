TRACES
[]*ResourceSpans{
    SchemaUrl string
    DeprecatedScopeSpans []*ScopeSpans ???
    Resource v1.Resource{
        DroppedAttributesCount uint32
        Attributes []11.KeyValue
    }
    ScopeSpans []*ScopeSpans{
        SchemaUrl string
        Scope v11.InstrumentationScope{
            Name string
            Version string
            DroppedAttributesCount uint32
            Attributes []v11.KeyValue
        }
        Spans []*Span{
            TraceId 16-Byte
            SpanId 8-Byte
            TraceState string
            ParentSpanId 8-Byte
            Name string
            Kind string (really an enum of Span_SpanKind that gets converted to a string)
            StartTimeUnixNano uint64
            EndTimeUnixNano uint64
            DroppedAttributesCount uint32
            Attributes []v11.KeyValue
            DroppedEventsCount uint32
            DroppedLinksCount uint32
            Status Status{
                Message string
                Code string (really an enum of Status_StatusCode that gets converted to a string)
            }
            Events []*Span_Event{
                TimeUnixNano uint64
                Name string
                DroppedAttributesCount uint32
                Attributes v11.KeyValue
            }
            Links []*Span_Link{
                TraceId 16-Byte
                SpanId 8-Byte
                TraceState string
                DroppedAttributesCount uint32
                Attributes v11.KeyValue
            }
        }
    }
}
